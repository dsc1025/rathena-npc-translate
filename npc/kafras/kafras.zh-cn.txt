		mes "被保存在这里，里面";

		mes "[卡夫拉・莱拉]";
		mes "请利用";
		mes "Kafra 服务是";
		mes "可以在所有的";
		mes "米德加德。谢谢你";
		mes "参观卡夫拉总部。";
		close2;
		break;
	case 2:
		if (Zeny < 20) {
			mes "[卡夫拉・莱拉]";
			mes "对不起，但是它";
			mes "看来你不";
			mes "有 20 zeny 需要支付";
			mes "存储访问费...";
			close3;
		}
		mes "[卡夫拉・莱拉]";
		mes "虽然这个设施是";
		mes "专门用于";
		mes "Kafra员工培训";
		mes "和行政职能，";
		mes "我将为您访问您的存储空间。";
		next;
		set Zeny, Zeny-20;
		set RESRVPTS, RESRVPTS + (20/5);
		mes "[卡夫拉・莱拉]";
		mes "未来，请";
		mes "询问 Kafra 员工";
		mes "如果你想使用的话";
		mes "任何 Kafra 服务。";
		mes "感谢您的惠顾。";
		callfunc("F_CheckKafCode");    //check your storage password, if set
		close2;
		openstorage;
		break;
	case 3:
		mes "[卡夫拉・莱拉]";
		mes "我很抱歉，但我";
		mes "不值班。我会帮助你";
		mes "如果我可以，但实际上没有";
		mes "有任何可用的手推车。";
		mes "你为什么不问另一个卡夫拉";
		mes "员工寻求帮助？";
		next;
		mes "[卡夫拉・莱拉]";
		mes "卡夫拉员工是";
		mes "驻扎在各地";
		mes "米德加尔特大陆，";
		mes "你应该能够找到";
		mes "Al De Baran 外面有很多。";
		close2;
		break;
	case 4:
		mes "[卡夫拉・莱拉]";
		mes "^666666*呼...*^000000";
		mes "太好了，因为我是";
		mes "实际上我在休息";
		mes "现在。选择";
		mes "“取消”是个好主意";
		mes "移动你的部分。";
		close2;
		break;
	}
	cutin "",255;
	end;
}
