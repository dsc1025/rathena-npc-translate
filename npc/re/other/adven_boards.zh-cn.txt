function	script	F_AvenBoard	{
	.@mo_board = select( "按楼层查找怪物",
					"Monster Information Retrieval System",
					( countitem(909) == 30000 ? "Request" : "" ),
					"Cancel" );
	if (.@mo_board == 1)
		return 1;
	else if (.@mo_board == 2) {
		while(1) {
			mes "[怪物信息检索系统]";
			mes "这就是怪物信息检索系统。可以搜索该迷宫中出现的怪物的信息。搜索时输入怪物名称，包括空格。";
			next;
			input .@inputstr$;
			setarray .@element$[0],"Neutral","Water","Earth","Fire","Wind","Poison","Holy","Shadow","Ghost","Undead";
			setarray .@size$[0],"Medium","Small","Big";
			setarray .@race$[0],"Formless","Undead","Animal","Plant","Insect","Fish","Demon","Demihuman","Angel","Dragon";
			.@size = getargcount();
			.@i = 0;
			while( .@i < .@size && getarg(.@i+1,"") != .@inputstr$ ) .@i += 3;
			mes "[怪物信息检索系统]";
			mes "现在正在寻找";
			next;
			mes "[怪物信息检索系统]";
			mes "搜索已完成。";
			mes "现在显示搜索结果。";
			next;
			mes "[怪物信息检索系统]";
			if (.@i >= .@size) {
				mes "没有关于该怪物的信息。";
				mes "请检查您的怪物名称。";
			}
			else {
				.@mob_id = getarg(.@i);
				mes "- 姓名 ："+ getarg(.@i+1);
				mes "- 元素 ："+ .@element$[ getmonsterinfo(.@mob_id,MOB_ELEMENT) ];
				mes "- 尺寸 ："+ .@size$[ getmonsterinfo(.@mob_id,MOB_SIZE) ];
				mes "- 种族 ："+ .@race$[ getmonsterinfo(.@mob_id,MOB_RACE) ];
				mes "--------------------------------^FD8D02";
				mes getarg(.@i+2);
			}
			next;
			mes "[怪物信息检索系统]";
			mes "您想搜索其他怪物信息吗？";
			next;
			if (select( "搜索","取消" ) == 1)
				continue;
			mes "[怪物信息检索系统]";
			mes "怪物搜索系统现在将关闭。";
			next;
			return 2;
		}
	}
	else if (.@mo_board == 3) {
		mes "会出现请求的信息 - 如果出现就没有问题";
		next;
		return 3;
	}
	else if (.@mo_board == 4)
		close;
	else {
		mes "操作错误。";
		close;
	}
}

	next;
	return;
}
