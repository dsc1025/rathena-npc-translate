-	script	ZondaTeleportationServices	HIDDEN_NPC,{
	cutin "zonda_do01", 2;
	mes "[Zonda员工]";
	mes "欢迎。";
	mes "我们的团队始终致力于超越您对服务质量的期望。";
	mes "请选择您的目的地。";
	next;
	switch( select( "北比弗罗斯特", "Eclage", "Splendide", "Midgard探险营地", "马努克", "卡米达尔山郊区", "取消。") ) {
	case 1:
		callsub( S_Zone, "North Bifrost", 17000,850, 7213 ); // Teleport value not confirmed
		warp "bif_fild02",160,230; // Unverified Coordinates
		end;
	case 2:
		callsub( S_Zone, "Eclage", 17000,850, 11311 );
		warp "eclage",111,39;
		end;
	case 3:
		callsub( S_Zone, "Splendide", 17000,850, 10078 );
		warp "splendide",171,115;
		end;
	case 4:
		callsub( S_Zone, "Midgard Expedition Camp", 19000,950, 10078 );
		warp "mid_camp",210,288;
		end;
	case 5:
		callsub( S_Zone, "Manuk", 22000,1100, 10078 );
		warp "manuk",263,177;
		end;
	case 6:
		callsub( S_Zone, "Outskirts of Kamidal Mountain", 24000,1200, 7184 );
		warp "dic_fild01",188,187;
		end;
	case 7:
		mes "[Zonda员工]";
		mes "我们始终致力于确保为客户提供最好的服务。";
		mes "感谢您使用我们的服务。";
		close3;
	}

S_Zone:
	.@zeny_req = getarg(1);
	.@item_req = getarg(2);
	.@quest_req = getarg(3);
	
	if (isbegin_quest(.@quest_req) == 0) {
		mes "[Zonda员工]";
		//mes getarg(0);
		mes "在使用我们的传送服务之前，您必须完成传送到" + getarg(0) + "的访问任务。";
		close3;
	}
	mes "[Zonda员工]";
	mes getarg(0) + "适用于 ^ff0000" + .@zeny_req + " Zenys^000000 或 ^ff0000" + .@item_req + " Malang Sp Can^000000。";
	next;
	switch( select( .@zeny_req + "女性", .@item_req + "玛琅Sp Can", "取消" ) ) {
	case 1:
		if (Zeny < .@zeny_req)
			break;
		Zeny -= .@zeny_req;
		return;
	case 2:
		if (countitem(12636) < .@item_req)
			break;
		delitem 12636,.@item_req; // Malang_Sp_Can
		return;
	case 3:
		mes "[Zonda员工]";
		mes "您能告诉我其他服务领域吗？";
		close3;
	}
	mes "[Zonda员工]";
	mes "先生！您没有提供我们所要求的足够价值。";
	mes "请再次回来查看。";
	close3;
}
